系统架构

PC 端画板用鼠标绘制图形 → 实时转为 [draw,...] 协议通过串口发送 → STM32 端解析 → 渲染到物理 OLED/LCD 屏幕。

方向协议用途
📥 PC→MCU[draw,指令,参数...]PC 画板绘制的图形 → MCU 解析 → 驱动 OLED/LCD
📤 MCU→PC[draw,指令,参数...]MCU 端生成图形 → PC 端 OLED 虚拟屏显示
📥 PC→MCU[display,x,y,text,size]旧协议，纯文本渲染（建议迁移到 draw,text）

源码位置：

  • MCU 解析器：oled_draw_test.c
  • MCU 生成器：draw_out.c
  • 虚拟屏 UI：test_display.c
  • OLED 驱动：OLED.c
  • LCD 驱动：lcd.c

快速上手

最简单的验证：PC 画板画个圆，MCU OLED 上显示出来。

MCU 端（三行初始化）

#include "Serial.h"
#include "OLED.h"

void main(void) {
    // ... CubeMX 生成的初始化 ...

    OLED_Init();                                                       // ① 初始化 OLED
    UART_InitReceive(SERIAL_DEVICE_1, SERIAL_PROTOCOL,
                     SERIAL_SQUARE_BRACKET);                          // ② 启动方括号协议

    while (1) {
        if (UART_GetRxFlag(SERIAL_DEVICE_1)) {
            char *raw = UART_GetRxData(SERIAL_DEVICE_1);              // ③ 取出消息
            DrawTest_ProcessCommand(raw);                             // ← 调用解析器
        }
    }
}
PC 端操作：切换到 OLED 绘图标签页 → 选圆形工具 → 在画布上拖一个圆 → MCU OLED 上实时显示。

全部绘图指令

协议外壳：[draw,指令,参数...]。所有坐标以像素为单位。

指令格式说明
点[draw,point,x,y,#color]单像素点
线[draw,line,x1,y1,x2,y2,#color,w]线段，w=线宽（px）
矩形[draw,rect,x,y,w,h,#color,w,fill]x,y=左上角；fill="fill" 则填充
圆角矩形[draw,rrect,x,y,w,h,#color,w,r,fill]r=圆角半径
圆[draw,circle,cx,cy,r,#color,w,fill]cx,cy=圆心
椭圆[draw,ellipse,cx,cy,rx,ry,#color,w,fill]rx,ry=半轴
三角形[draw,triangle,x1,y1,x2,y2,x3,y3,#color,w,fill]三个顶点
弧线[draw,arc,cx,cy,r,start,end,#color,w,fill]start,end=角度（度）
文字[draw,text,x,y,content,size,#color]size 映射 OLED 字号
清屏[draw,clear,#color]可选背景色
填充矩形[draw,fill,x,y,w,h,#color]纯色填充（LCD 专用）

可选参数规则

w（线宽）和 fill 均为可选。解析器在两个可能的字段位置检查 "fill" 字符串：

// circle 示例：field[4] = r, field[5] = color
// w 在 field[6]（如果存在），fill 在 field[6] 或 field[7]
char fill_flag[8];
GetField(cmd, 6, fill_flag, sizeof(fill_flag));
int filled = (strcmp(fill_flag, "fill") == 0);
if (!filled) {
    GetField(cmd, 7, fill_flag, sizeof(fill_flag));  // 再试 field[7]
    filled = (strcmp(fill_flag, "fill") == 0);
}

MCU 端解析 — GetField

所有 draw 协议解析的第一步：从逗号串中拆出字段。下面是可以直接复制的完整实现。

/**
 * @brief  从逗号分隔的字符串中提取第 field 个字段（从 0 开始）
 * @param  src      源字符串，如 "circle,64,32,20,#FFF,fill"
 * @param  dst      输出缓冲区
 * @param  field    字段序号（0 = 第一个逗号之前）
 * @param  max_len  输出缓冲区大小
 * @retval 0=成功，-1=字段不存在
 *
 * @note   小数点 '.' 不是分隔符——"42.5" 不会被拆断
 *         dst 为空串 "" 表示字段不存在或越界
 */
int GetField(const char *src, char *dst, int field, int max_len) {
    const char *start = src;
    for (int i = 0; i < field; i++) {
        const char *comma = strchr(start, ',');
        if (!comma) { dst[0] = '\0'; return -1; }
        start = comma + 1;
    }
    const char *end = strchr(start, ',');
    int len = end ? (int)(end - start) : (int)strlen(start);
    if (len >= max_len) len = max_len - 1;
    memcpy(dst, start, len);
    dst[len] = '\0';
    return 0;
}

/* ── 辅助宏 ── */
#define FIELD_STR(i)  (GetField(cmd, i, buf, sizeof(buf)) == 0 ? buf : "")
#define FIELD_INT(i, def) (GetField(cmd, i, buf, sizeof(buf)) == 0 ? atoi(buf) : (def))

解析示例

// 输入：cmd = "circle,64,32,20,#FF0000,2,fill"
//                   0      1  2  3   4      5  6

char buf[32];
GetField(cmd, 0, buf, sizeof(buf));  // buf = "circle"
GetField(cmd, 1, buf, sizeof(buf));  // buf = "64"
GetField(cmd, 3, buf, sizeof(buf));  // buf = "20"
GetField(cmd, 6, buf, sizeof(buf));  // buf = "fill"

MCU 端解析 — type 路由

收到一条方括号消息后，先取第 0 个字段判断是哪种图形，再交给对应 handler。

void DrawTest_ProcessCommand(const char *raw) {
    char type[16];
    GetField(raw, 0, type, sizeof(type));

    if      (strcmp(type, "clear")    == 0) Handle_Clear(raw);
    else if (strcmp(type, "point")    == 0) Handle_Point(raw);
    else if (strcmp(type, "line")     == 0) Handle_Line(raw);
    else if (strcmp(type, "rect")     == 0) Handle_Rect(raw);
    else if (strcmp(type, "rrect")    == 0) Handle_RoundedRect(raw);
    else if (strcmp(type, "circle")   == 0) Handle_Circle(raw);
    else if (strcmp(type, "ellipse")  == 0) Handle_Ellipse(raw);
    else if (strcmp(type, "triangle") == 0) Handle_Triangle(raw);
    else if (strcmp(type, "arc")      == 0) Handle_Arc(raw);
    else if (strcmp(type, "text")     == 0) Handle_Text(raw);
    else if (strcmp(type, "fill")     == 0) Handle_FillRect(raw);
    else if (strcmp(type, "set")      == 0) Handle_Set(raw);      // F5
    else if (strcmp(type, "del")      == 0) Handle_Del(raw);      // F5
}

MCU 端解析 — 图形处理（完整代码）

以下以 LCD（彩色）为基准——带颜色解析、坐标钳位、旋转检测。OLED 单色版更简单，差异处已标注。

颜色解析（LCD 专用）

/* "#RRGGBB" → RGB565（16 位色） */
static uint8_t HexNibble(char c) {
    if (c >= '0' && c <= '9') return c - '0';
    if (c >= 'A' && c <= 'F') return c - 'A' + 10;
    if (c >= 'a' && c <= 'f') return c - 'a' + 10;
    return 0;
}
static uint16_t ParseColor(const char *s) {
    if (!s || s[0] != '#') return 0xFFFF;  // 默认白色
    uint8_t r = (HexNibble(s[1]) << 4) | HexNibble(s[2]);
    uint8_t g = (HexNibble(s[3]) << 4) | HexNibble(s[4]);
    uint8_t b = (HexNibble(s[5]) << 4) | HexNibble(s[6]);
    return ((uint16_t)(r >> 3) << 11) | ((uint16_t)(g >> 2) << 5) | (b >> 3);
}
/* OLED 注意：无颜色，用 IsEraserColor 检测暗色 = 橡皮擦 */

坐标钳位 + fill 检测

#define LCD_W 240
#define LCD_H 280

static int16_t ClampX(int16_t x) { return (x < 0) ? 0 : (x >= LCD_W) ? LCD_W-1 : x; }
static int16_t ClampY(int16_t y) { return (y < 0) ? 0 : (y >= LCD_H) ? LCD_H-1 : y; }

static int IsFillFlag(const char *s) {
    return (s && strcmp(s, "fill") == 0);
}
/* OLED 注意：分辨率 128×64，无需钳位 */

点 / 线

static void Handle_Point(const char *cmd) {
    int x = ClampX(FIELD_INT(1, 0)), y = ClampY(FIELD_INT(2, 0));
    char c[10]; GetField(cmd, c, 3, sizeof(c));
    uint16_t color = ParseColor(c);
    LCD_DrawPoint(x, y, color);                        // OLED: OLED_DrawPoint(x, y)
}

static void Handle_Line(const char *cmd) {
    int x1 = ClampX(FIELD_INT(1,0)), y1 = ClampY(FIELD_INT(2,0));
    int x2 = ClampX(FIELD_INT(3,0)), y2 = ClampY(FIELD_INT(4,0));
    char c[10]; GetField(cmd, c, 5, sizeof(c));
    uint16_t color = ParseColor(c);
    LCD_DrawLine(x1, y1, x2, y2, color);              // OLED: OLED_DrawLine(x1,y1,x2,y2)
}

矩形（含旋转检测）

static void Handle_Rect(const char *cmd) {
    int x = FIELD_INT(1,0), y = FIELD_INT(2,0), w = FIELD_INT(3,0), h = FIELD_INT(4,0);
    if (w < 1) w = 1; if (h < 1) h = 1;
    char c[10]; GetField(cmd, c, 5, sizeof(c));
    uint16_t color = ParseColor(c);

    /* fill 在 field 6 或 7（线宽可能占 field 6） */
    char f[8]; int filled = 0;
    if ((GetField(cmd, f, 6, sizeof(f))==0 && IsFillFlag(f)) ||
        (GetField(cmd, f, 7, sizeof(f))==0 && IsFillFlag(f))) filled = 1;

    /* 旋转检测：末尾字段以 'a' 开头，如 a45 */
    const char *p = cmd, *last = cmd;
    while (*p) { if (*p == ',') last = p + 1; p++; }
    int angle = (last[0] == 'a') ? atoi(last + 1) : 0;

    if (angle != 0) {
        LCD_DrawRotatedRect(x, y, w, h, angle, color, filled, 1);
        return;
    }
    int xe = ClampX(x + w - 1), ye = ClampY(y + h - 1);
    x = ClampX(x); y = ClampY(y);
    if (filled) LCD_Fill(x, y, xe, ye, color);
    else        LCD_DrawRectangle(x, y, xe, ye, color);
}
/* OLED: 同上但无颜色，OLED_DrawRectangle(x,y,w,h,filled)，OLED_DrawRotatedRect 无 color/lw */

圆 / 三角形

static void Handle_Circle(const char *cmd) {
    int cx = FIELD_INT(1,0), cy = FIELD_INT(2,0), r = FIELD_INT(3,0);
    if (r < 1) r = 1;
    char c[10]; GetField(cmd, c, 4, sizeof(c));
    uint16_t color = ParseColor(c);

    char f[8]; int filled = 0;
    if ((GetField(cmd, f, 5, sizeof(f))==0 && IsFillFlag(f)) ||
        (GetField(cmd, f, 6, sizeof(f))==0 && IsFillFlag(f))) filled = 1;

    if (filled) LCD_DrawFilledCircle(cx, cy, r, color);  /* 扫描线填充 */
    else        Draw_Circle(cx, cy, r, color);
}
/* OLED: 同上，去颜色：OLED_DrawCircle(cx,cy,r,filled) */

static void Handle_Triangle(const char *cmd) {
    int x0 = FIELD_INT(1,0), y0 = FIELD_INT(2,0);
    int x1 = FIELD_INT(3,0), y1 = FIELD_INT(4,0);
    int x2 = FIELD_INT(5,0), y2 = FIELD_INT(6,0);
    char c[10]; GetField(cmd, c, 7, sizeof(c));
    uint16_t color = ParseColor(c);

    char f[8]; int filled = 0;
    if ((GetField(cmd, f, 8, sizeof(f))==0 && IsFillFlag(f)) ||
        (GetField(cmd, f, 9, sizeof(f))==0 && IsFillFlag(f))) filled = 1;

    if (filled) LCD_DrawFilledTriangle(x0,y0, x1,y1, x2,y2, color);
    else {
        LCD_DrawLine(ClampX(x0),ClampY(y0), ClampX(x1),ClampY(y1), color);
        LCD_DrawLine(ClampX(x1),ClampY(y1), ClampX(x2),ClampY(y2), color);
        LCD_DrawLine(ClampX(x2),ClampY(y2), ClampX(x0),ClampY(y0), color);
    }
}
/* OLED: 同上，去颜色：OLED_DrawTriangle(x0,y0,x1,y1,x2,y2,filled) */

文字 / 清屏

static void Handle_Text(const char *cmd) {
    int x = ClampX(FIELD_INT(1,0)), y = ClampY(FIELD_INT(2,0));
    int fontSize = FIELD_INT(4, 16);
    char content[128]; GetField(cmd, content, 3, sizeof(content));
    char c[10]; GetField(cmd, c, 5, sizeof(c));
    uint16_t color = ParseColor(c);

    /* PC 字号 → LCD 字号 */
    uint8_t sizey;
    if (fontSize <= 12)      sizey = 12;
    else if (fontSize <= 16) sizey = 16;
    else if (fontSize <= 24) sizey = 24;
    else                      sizey = 32;
    LCD_ShowString(x, y, (uint8_t*)content, color, BLACK, sizey, 0);
}
/* OLED: OLED_ShowString(x, y, str, font)，font 只有 6×8 / 8×16 两档 */

static void Handle_Clear(const char *cmd) {
    char c[10] = {0}; uint16_t bg = BLACK;
    if (GetField(cmd, c, 1, sizeof(c))==0 && c[0]=='#') bg = ParseColor(c);
    f5_bg = bg;  f5_count = 0;   /* F5 复位 */
    LCD_ClearFast(bg);           /* OLED: OLED_Clear() + OLED_Update() */
}
📌 复杂图形（椭圆 / 圆角矩形 / 弧线）：LCD 驱动不提供这些函数，handler 里从零实现了 Bresenham 圆角矩形（~60 行）、中点椭圆算法（~30 行）、atan2f 弧线角度过滤（~40 行）。完整源码见 lcd_draw_test.c。OLED 驱动自带全部复杂图形函数，handler 就是简单转发。

F5 增量同步 — 原理

PC 画板上拖动一个图形时，如果每次都发全量重绘（几十条 draw 命令），串口带宽吃不消。F5 机制：PC 只发变化的那个图形，MCU 端本地维护一份图形数组，收到新数据后本地全量重绘。

步骤PC 发送MCU 动作
新增图形[draw,set,a1,circle,64,32,20,#0F0]存入 f5_cmds[0]，重绘全部
修改图形[draw,set,a1,circle,64,32,25,#0F0]同 ID 覆盖（半径 20→25），重绘全部
删除图形[draw,del,a1]移除 f5_cmds[0]，前移数组，重绘全部
拖动过程只发 [draw,set,a1,...]每次只更新 1 条 → 1 次全量重绘

关键：每帧串口只需传 1 条 set/del 消息，不需要把整个画布的所有图形重发一遍。

F5 增量同步 — 完整代码

#define F5_MAX_SHAPES 32

static char  f5_ids[F5_MAX_SHAPES][8];        // 图形 ID，如 "a1", "r5"
static char  f5_cmds[F5_MAX_SHAPES][256];     // 存储的 draw 命令（不含 "draw," 前缀）
static int   f5_count = 0;

/* ── 按 ID 查找 ── */
static int f5_find(const char *id) {
    for (int i = 0; i < f5_count; i++)
        if (strcmp(f5_ids[i], id) == 0) return i;
    return -1;
}

/* ── 全量重绘 ── */
static void f5_redraw_all(void) {
    OLED_Clear();
    for (int i = 0; i < f5_count; i++) {
        char full[280];
        snprintf(full, sizeof(full), "draw,%s", f5_cmds[i]);
        DrawTest_ProcessCommand(full);          // 复用路由解析器
    }
    OLED_Update();
}

/* ── set 处理：新增或更新图形 ── */
static void Handle_Set(const char *cmd) {
    // 协议：set,id,type,params...
    char id[8];
    GetField(cmd, 1, id, sizeof(id));

    // 提取 "draw," 之后的部分：跳过 "set,id,"
    const char *rest = strchr(cmd, ',');
    if (rest) rest = strchr(rest + 1, ',');  // 跳过 id 字段
    if (!rest) return;
    rest++;  // 指向 type

    int idx = f5_find(id);
    if (idx < 0) {
        if (f5_count >= F5_MAX_SHAPES) return;
        idx = f5_count++;
        strcpy(f5_ids[idx], id);
    }
    strncpy(f5_cmds[idx], rest, sizeof(f5_cmds[idx]) - 1);
    f5_redraw_all();
}

/* ── del 处理：删除图形 ── */
static void Handle_Del(const char *cmd) {
    char id[8];
    GetField(cmd, 1, id, sizeof(id));

    int idx = f5_find(id);
    if (idx < 0) return;

    // 前移数组
    for (int i = idx; i < f5_count - 1; i++) {
        strcpy(f5_ids[i], f5_ids[i + 1]);
        strcpy(f5_cmds[i], f5_cmds[i + 1]);
    }
    f5_count--;
    f5_redraw_all();
}

旋转支持

矩形、圆角矩形、椭圆支持旋转。协议末尾加 a<angle>（角度制）：

[draw,rect,200,150,100,60,#FF0000,2,a45]

MCU 端用旋转矩阵计算四个角点：

static void LCD_DrawRotatedRect(int x, int y, int w, int h,
                                float angle, uint16_t color, int filled) {
    float rad = angle * 3.14159f / 180.0f;
    float cos_a = cosf(rad), sin_a = sinf(rad);

    // 四个角（相对于矩形左上角）
    int dx[] = {0, w, w, 0}, dy[] = {0, 0, h, h};
    int px[4], py[4];

    for (int i = 0; i < 4; i++) {
        // 旋转矩阵
        px[i] = x + (int)(dx[i] * cos_a - dy[i] * sin_a);
        py[i] = y + (int)(dx[i] * sin_a + dy[i] * cos_a);
    }

    if (filled) {
        // 填充旋转矩形：用三角形分解
        LCD_DrawFilledTriangle(px[0],py[0], px[1],py[1], px[2],py[2], color);
        LCD_DrawFilledTriangle(px[0],py[0], px[2],py[2], px[3],py[3], color);
    } else {
        // 轮廓：四条线段
        for (int i = 0; i < 4; i++) {
            int j = (i + 1) % 4;
            LCD_DrawLine(px[i], py[i], px[j], py[j], color);
        }
    }
}

旋转椭圆的实现类似——对每个扫描线点应用旋转矩阵后填充或描边。

橡皮擦颜色

OLED 是单色的，无法"画个白色矩形"来模拟橡皮擦。PC 画板的橡皮擦工具发送特定暗色——MCU 端识别这些颜色后，对 line 执行清除逻辑，对其他形状跳过绘制。

/* 橡皮擦颜色检测 */
static int IsEraserColor(const char *color_str) {
    if (strcmp(color_str, "#111111") == 0) return 1;
    if (strcmp(color_str, "#000000") == 0) return 1;
    // 匹配 #11xxxx / #22xxxx / #00xxxx 模式
    if (color_str[0] == '#' &&
        color_str[1] == color_str[2] &&
        (color_str[1] == '0' || color_str[1] == '1' || color_str[1] == '2'))
        return 1;
    return 0;
}

/* 在处理 line 时检测橡皮擦 */
static void Handle_Line(const char *cmd) {
    char color[8];
    GetField(cmd, 5, color, sizeof(color));

    if (IsEraserColor(color)) {
        // 执行清除线段（把像素设为 0）
        Handle_ClearLine(cmd);
        return;
    }
    // 正常绘制线段
    // ...
}
